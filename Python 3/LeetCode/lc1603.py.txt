class ParkingSystem:

    def __init__(self, big: int, medium: int, small: int):
        self.one = big
        self.two = medium
        self.three = small

    def addCar(self, carType: int) -> bool:
        if carType==1:
            if self.one>0:
                self.one-=1
                return True
            return False
        elif carType==2:
            if self.two>0:
                self.two-=1
                return True
            return False
        elif carType==3:
            if self.three>0:
                self.three-=1
                return True
            return False
